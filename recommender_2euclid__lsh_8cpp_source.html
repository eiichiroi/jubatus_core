52 float calc_norm(const common::sfv_t& sfv) {
53  float sqnorm = 0;

60 vector<float> lsh_function(
61  const common::sfv_t& query,
62  size_t dimension,
63  float bin_width) {
64  vector<float> hash(dimension);
65  for (size_t i = 0; i < query.size(); ++i) {
66  const uint32_t seed = common::hash_util::calc_string_hash(query[i].first);
67  mtrand rnd(seed);
68  for (size_t j = 0; j < hash.size(); ++j) {
69  hash[j] += query[i].second * rnd.next_gaussian();
70  }
71  }
72  for (size_t j = 0; j < dimension; ++j) {
73  hash[j] /= bin_width;
74  }
75  return hash;
76 }

230 vector<float> euclid_lsh::calculate_lsh(const common::sfv_t& query) {
231  vector<float> hash(mixable_storage_->get_model()->all_lsh_num());
232  for (size_t i = 0; i < query.size(); ++i) {
233  const uint32_t seed = common::hash_util::calc_string_hash(query[i].first);
234  const vector<float> proj = get_projection(seed);
235  for (size_t j = 0; j < hash.size(); ++j) {
236  hash[j] += query[i].second * proj[j];
237  }
238  }
239  for (size_t j = 0; j < hash.size(); ++j) {
240  hash[j] /= bin_width_;
241  }
242  return hash;
243 }

245 vector<float> euclid_lsh::get_projection(uint32_t seed) {
246  vector<float> tmpl_proj;
247  vector<float>& proj = retain_projection_ ? projection_[seed] : tmpl_proj;
248 
249  if (proj.empty()) {
250  mtrand rnd(seed);
251  proj.resize(mixable_storage_->get_model()->all_lsh_num());
252  for (size_t i = 0; i < proj.size(); ++i) {
253  proj[i] = rnd.next_gaussian();
254  }
255  }
256  return proj;
257 }
