85 void normalize(eigen_vector_diff& v) {
86  double sum = 0;
87  for (eigen_vector_diff::const_iterator it = v.begin(); it != v.end(); ++it) {
88  sum += it->second.score;
89  }
90  const double normalizer = v.size() / sum;
91  for (eigen_vector_diff::iterator it = v.begin(); it != v.end(); ++it) {
92  it->second.score *= normalizer;
93  }
94 }

283 void graph_wo_index::shortest_path(
284  node_id_t src,
285  node_id_t tgt,
286  uint64_t max_hop,
287  std::vector<node_id_t>& ret,
288  const preset_query& query) const {
289  if (global_nodes_.count(src) == 0) {
290  throw JUBATUS_EXCEPTION(unknown_id("shortest_path:src", src));
291  }
292  if (global_nodes_.count(tgt) == 0) {
293  throw JUBATUS_EXCEPTION(unknown_id("shortest_path:tgt", tgt));
294  }
295  spt_query_diff::const_iterator model_it = spts_.find(query);
296  if (model_it == spts_.end()) {
297  throw JUBATUS_EXCEPTION(unknown_query(query));
298  }
299  const spt_diff& mixed = model_it->second;
300  ret.clear();
301  uint64_t min_score = ~uint64_t();
302  uint64_t ind = ~uint64_t();
303  for (uint64_t i = 0; i < mixed.size(); ++i) {
304  const shortest_path_tree& spt = mixed[i];
305  spt_edges::const_iterator src_it = spt.to_root.find(src);
306  spt_edges::const_iterator tgt_it = spt.from_root.find(tgt);
307  if (src_it == spt.to_root.end() || tgt_it == spt.from_root.end()) {
308  continue;
309  }
310  uint64_t score = src_it->second.first + tgt_it->second.first;
311  if (score < min_score) {
312  min_score = score;
313  ind = i;
314  }
315  }
316 
317  if (ind >= static_cast<uint64_t>(config_.landmark_num)) {
318  return;
319  }
320 
321  const spt_edges& to_root = mixed[ind].to_root;
322  const spt_edges& from_root = mixed[ind].from_root;
323  uint64_t landmark = mixed[ind].landmark;
324 
325  for (uint64_t cur = src; cur != landmark;) {
326  if (ret.size() >= max_hop) {
327  return;
328  }
329  spt_edges::const_iterator it = to_root.find(cur);
330  if (it == to_root.end()) {
331  ret.clear();
332  return;
333  }
334  ret.push_back(cur);
335  if (cur == tgt) {
336  return;
337  }
338  cur = it->second.second;
339  }
340  ret.push_back(landmark);
341 
342  std::vector<node_id_t> from_root_path;
343  for (uint64_t cur = tgt; cur != landmark;) {
344  if (ret.size() + from_root_path.size() >= max_hop) {
345  return;
346  }
347  spt_edges::const_iterator it = from_root.find(cur);
348  if (it == from_root.end()) {
349  ret.clear();
350  return;
351  }
352  from_root_path.push_back(cur);
353  cur = it->second.second;
354  }
355 
356  for (vector<node_id_t>::const_reverse_iterator rit = from_root_path.rbegin();
357  rit != from_root_path.rend(); ++rit) {
358  ret.push_back(*rit);
359  }
360 }

397 void graph_wo_index::get_diff_eigen_score(eigen_vector_query_diff& diff) const {
398  diff.clear(); // tmp_diff + swap is better ?
399 
400  for (eigen_vector_query_diff::const_iterator query_it
401  = eigen_scores_.begin();
402  query_it != eigen_scores_.end(); ++query_it) {
403  const preset_query& query = query_it->first;
404  const eigen_vector_diff& model = query_it->second;
405 
406  double dist = 0;
407  for (eigen_vector_diff::const_iterator it = model.begin();
408  it != model.end(); ++it) {
409  if (it->second.out_degree_num == 0) {
410  dist += it->second.score;
411  }
412  }
413 
414  unordered_set<node_id_t> unmatched_nodes;
415 
416  uint64_t new_node_num = 0;
417  double dist_from_new_node = 0;
418  for (node_info_map::const_iterator node_it = local_nodes_.begin();
419  node_it != local_nodes_.end(); ++node_it) {
420  if (model.count(node_it->first) == 0) {
421  if (!is_matched_to_query(query.node_query, node_it->second.property)) {
422  unmatched_nodes.insert(node_it->first);
423  continue;
424  }
425  dist_from_new_node += 1.0;
426  ++new_node_num;
427  }
428  }
429  dist += dist_from_new_node;
430 
431  if (model.size() + new_node_num > 0) {
432  dist /= (model.size() + new_node_num);
433  }
434 
435  eigen_vector_diff& qdiff = diff[query];
436 
437  for (node_info_map::const_iterator node_it = local_nodes_.begin();
438  node_it != local_nodes_.end(); ++node_it) {
439  if (unmatched_nodes.count(node_it->first)) {
440  continue;
441  }
442  if (!is_matched_to_query(query.node_query, node_it->second.property)) {
443  unmatched_nodes.insert(node_it->first);
444  continue;
445  }
446 
447  const std::vector<edge_id_t>& in_edges = node_it->second.in_edges;
448  double score = 0;
449  for (size_t i = 0; i < in_edges.size(); ++i) {
450  if (unmatched_nodes.count(in_edges[i])) {
451  continue;
452  }
453  edge_info_map::const_iterator edge_it = local_edges_.find(in_edges[i]);
454  if (edge_it == local_edges_.end()) {
455  continue;
456  }
457 
458  const node_id_t src = edge_it->second.src;
459  if (unmatched_nodes.count(src)) {
460  continue;
461  }
462  if (!is_matched_to_query(query.edge_query, edge_it->second.p)) {
463  continue;
464  }
465 
466  eigen_vector_diff::const_iterator it = model.find(edge_it->second.src);
467  if (it == model.end()) {
468  continue;
469  }
470  if (it->second.out_degree_num != 0) {
471  // TODO(beam2d) it->second.score > 0 should indicate
472  // it->second.out_degree_num
473  score += it->second.score / it->second.out_degree_num;
474  }
475  }
476 
477  eigen_vector_info ei;
478  ei.score = config_.damping_factor * score + 1 - config_.damping_factor
479  + config_.damping_factor * dist;
480 
481  if (is_empty_query(query)) {
482  ei.out_degree_num = node_it->second.out_edges.size();
483  } else {
484  uint64_t out_degree = 0;
485  for (size_t i = 0; i < node_it->second.out_edges.size(); ++i) {
486  const edge_info_map::const_iterator edge_it
487  = local_edges_.find(node_it->second.out_edges[i]);
488  const edge_info& edge = edge_it->second;
489  if (unmatched_nodes.count(edge.tgt)) {
490  continue;
491  }
492  if (!is_node_matched_to_query(query, edge.tgt)) {
493  unmatched_nodes.insert(edge.tgt);
494  continue;
495  }
496  if (!is_matched_to_query(query.edge_query, edge.p)) {
497  continue;
498  }
499  ++out_degree;
500  }
501  ei.out_degree_num = out_degree;
502  }
503 
504  qdiff[node_it->first] = ei;
505  }
506  }
507 }

539 void graph_wo_index::update_spt_node(
540  const preset_query& query,
541  const std::vector<edge_id_t>& edges,
542  spt_edges& se,
543  bool is_out) {
544  for (size_t i = 0; i < edges.size(); ++i) {
545  edge_info& edge = local_edges_[edges[i]];
546  const node_id_t from = is_out ? edge.src : edge.tgt;
547  const node_id_t to = is_out ? edge.tgt : edge.src;
548 
549  if (!is_matched_to_query(query.edge_query, edge.p)
550  || !is_node_matched_to_query(query, from)
551  || !is_node_matched_to_query(query, to)) {
552  continue;
553  }
554 
555  spt_edges::iterator tr_it = se.find(from);
556  if (tr_it != se.end()) {
557  uint64_t dist = tr_it->second.first + 1;
558  spt_edges::iterator tr_jt = se.find(to);
559  if (tr_jt != se.end()) {
560  if (dist < tr_jt->second.first) {
561  tr_jt->second.first = dist;
562  tr_jt->second.second = from;
563  }
564  } else {
565  se.insert(std::make_pair(to, std::make_pair(dist, from)));
566  }
567  }
568  }

596  for (spt_query_diff::const_iterator it = spts_.begin(); it != spts_.end();
597  ++it) {

619  if (to_it != spt.to_root.end()) {

684 void graph_wo_index::mix(
685  const spt_query_diff& all_diff,
686  spt_query_diff& all_mixed,
687  size_t landmark_num) {
688  for (spt_query_diff::const_iterator it = all_diff.begin();
689  it != all_diff.end(); ++it) {
690  const spt_diff& diff = it->second;
691  spt_diff& mixed = all_mixed[it->first];
692 
693  map<node_id_t, uint64_t> diff_landmark2ind;
694  for (uint64_t i = 0; i < diff.size(); ++i) {
695  if (diff[i].landmark != LLONG_MAX) {
696  diff_landmark2ind[diff[i].landmark] = i;
697  }
698  }
699 
700  map<node_id_t, uint64_t> mixed_landmark2ind;
701  for (uint64_t i = 0; i < mixed.size(); ++i) {
702  if (mixed[i].landmark != LLONG_MAX) {
703  mixed_landmark2ind[mixed[i].landmark] = i;
704  }
705  }
706 
707  for (map<node_id_t, uint64_t>::const_iterator it
708  = diff_landmark2ind.begin(); it != diff_landmark2ind.end(); ++it) {
709  map<node_id_t, uint64_t>::iterator jt
710  = mixed_landmark2ind.find(it->first);
711  if (jt != mixed_landmark2ind.end()) {
712  mix_spt(diff[it->second], mixed[jt->second]);
713  } else if (mixed.size() < landmark_num) {
714  mixed.push_back(diff[it->second]);
715  }
716  }
717  }
