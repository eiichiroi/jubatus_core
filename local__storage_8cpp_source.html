74 void local_storage::inp(const common::sfv_t& sfv, map_feature_val1_t& ret)
75  const {
76  ret.clear();
77 
78  // Use uin64_t map instead of string map as hash function for string is slow
79  jubatus::util::data::unordered_map<uint64_t, float> ret_id;
80  for (common::sfv_t::const_iterator it = sfv.begin(); it != sfv.end(); ++it) {
81  const string& feature = it->first;
82  const float val = it->second;
83  id_features3_t::const_iterator it2 = tbl_.find(feature);
84  if (it2 == tbl_.end()) {
85  continue;
86  }
87  const id_feature_val3_t& m = it2->second;
88  for (id_feature_val3_t::const_iterator it3 = m.begin(); it3 != m.end();
89  ++it3) {
90  ret_id[it3->first] += it3->second.v1 * val;
91  }
92  }
93 
94  std::vector<std::string> labels = class2id_.get_all_id2key();
95  for (size_t i = 0; i < labels.size(); ++i) {
96  const std::string& label = labels[i];
97  uint64_t id = class2id_.get_id_const(label);
98  if (id == common::key_manager::NOTFOUND || ret_id.count(id) == 0) {
99  ret[label] = 0.0;
100  } else {
101  ret[label] = ret_id[id];
102  }
103  }
104 }

144 void local_storage::bulk_update(
145  const common::sfv_t& sfv,
146  float step_width,
147  const string& inc_class,
148  const string& dec_class) {
149  uint64_t inc_id = class2id_.get_id(inc_class);
150  typedef common::sfv_t::const_iterator iter_t;
151  if (dec_class != "") {
152  uint64_t dec_id = class2id_.get_id(dec_class);
153  for (iter_t it = sfv.begin(); it != sfv.end(); ++it) {
154  float val = it->second * step_width;
155  id_feature_val3_t& feature_row = tbl_[it->first];
156  feature_row[inc_id].v1 += val;
157  feature_row[dec_id].v1 -= val;
158  }
159  } else {
160  for (iter_t it = sfv.begin(); it != sfv.end(); ++it) {
161  float val = it->second * step_width;
162  id_feature_val3_t& feature_row = tbl_[it->first];
163  feature_row[inc_id].v1 += val;
164  }
165  }
166 }

191 bool local_storage::delete_label(const std::string& label) {
192  uint64_t delete_id = class2id_.get_id_const(label);
193  if (delete_id == common::key_manager::NOTFOUND) {
194  return false;
195  }
196  for (id_features3_t::iterator it = tbl_.begin();
197  it != tbl_.end();
198  ) {
199  const bool deleted = it->second.erase(delete_id);
200  if (deleted && it->second.empty()) {
201  it = tbl_.erase(it);
202  } else {
203  ++it;
204  }
205  }
206  class2id_.delete_key(label);
207  return true;
208 }
