50 template <typename T>
51 void partial_sort_by(
52  const std::vector<double>& scores,
53  std::vector<T>& array,
54  size_t size) {
55  JUBATUS_ASSERT_EQ(scores.size(),
56  array.size(),
57  "lengths of scores and data must be same.");
58 
59  std::vector<std::pair<double, T> > pairs(scores.size());
60  for (size_t i = 0; i < scores.size(); ++i) {
61  pairs[i].first = scores[i];
62  swap(pairs[i].second, array[i]);
63  }
64  size = std::min(size, pairs.size());
65  std::partial_sort(pairs.begin(),
66  pairs.begin() + size,
67  pairs.end(),
68  compare_by_first());
69  array.resize(size);
70  for (size_t i = 0; i < size; ++i) {
71  swap(pairs[i].second, array[i]);
72  }
73 }

135 void kmeans_compressor::get_bicriteria(
136  const wplist& src,
137  size_t bsize,
138  size_t dstsize,
139  wplist& dst) {
140  dst.clear();
141  wplist resid = src;
142  vector<double> weights(src.size());
143  double r =
144  (1 - std::exp(bsize * (std::log(bsize) - std::log(src.size())) / dstsize))
145  / 2;
146  r = max(0.1, r);
147  std::vector<size_t> ind(bsize);
148  while (resid.size() > 1 && dst.size() < dstsize) {
149  weights.resize(resid.size());
150  for (wplist::const_iterator it = resid.begin(); it != resid.end(); ++it) {
151  weights[it - resid.begin()] = it->weight;
152  }
153 
154  // Sample `bsize` points and insert them to the result
155  discrete_distribution d(weights.begin(), weights.end());
156  std::generate(ind.begin(), ind.end(), d);
157 
158  std::sort(ind.begin(), ind.end());
159  ind.erase(std::unique(ind.begin(), ind.end()), ind.end());
160 
161  for (std::vector<size_t>::iterator it = ind.begin();
162  it != ind.end(); ++it) {
163  dst.push_back(resid[*it]);
164  }
165 
166  // Remove `r` nearest points from `resid`
167  std::vector<double> distances;
168  for (wplist::iterator itr = resid.begin(); itr != resid.end(); ++itr) {
169  distances.push_back(-min_dist(*itr, dst).second);
170  }
171  // TODO(unno): Is `r` lesser than 1.0?
172  size_t size = std::min(resid.size(),
173  static_cast<size_t>(resid.size() * r));
174  if (size == 0) {
175  size = 1;
176  }
177 
178  partial_sort_by(distances, resid, size);
179  }
180 }

193 void kmeans_compressor::bicriteria_to_coreset(
194  const wplist& src,
195  const wplist& bicriteria,
196  size_t dstsize,
197  wplist& dst) {
198  if (bicriteria.size() == 0) {
199  dst = src;
200  return;
201  }
202  double weight_sum = 0;
203  double squared_min_dist_sum = 0;
204  std::vector<size_t> nearest_indexes(src.size());
205  std::vector<double> nearest_distances(src.size());
206  std::vector<double> bicriteria_scores(bicriteria.size());
207  for (size_t i = 0; i < src.size(); ++i) {
208  double weight = src[i].weight;
209  std::pair<int, double> m = min_dist(src[i], bicriteria);
210  nearest_indexes[i] = m.first;
211  nearest_distances[i] = m.second;
212 
213  bicriteria_scores[m.first] += weight;
214  squared_min_dist_sum += m.second * m.second * weight;
215  weight_sum += weight;
216  }
217  std::vector<double> weights;
218  double sumw = 0;
219  for (size_t i = 0; i < src.size(); ++i) {
220  double prob = get_probability(
221  src[i],
222  nearest_distances[i],
223  bicriteria[nearest_indexes[i]],
224  bicriteria_scores[nearest_indexes[i]],
225  weight_sum,
226  squared_min_dist_sum);
227  weights.push_back(prob);
228  sumw += prob;
229  }
230 
231  discrete_distribution d(weights.begin(), weights.end());
232 
233  std::vector<size_t> ind(dstsize);
234  std::generate(ind.begin(), ind.end(), d);
235 
236  for (std::vector<size_t>::iterator it = ind.begin(); it != ind.end(); ++it) {
237  size_t index = *it;
238  weighted_point sample = src[index];
239  double prob = weights[index] / sumw;
240  sample.weight *= 1.0 / dstsize / prob;
241  dst.push_back(sample);
242  }
243 }
