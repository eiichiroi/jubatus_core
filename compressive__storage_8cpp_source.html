87 void compressive_storage::carry_up(size_t r) {
88  if (r >= mine_.size() - 1) {
89  mine_.push_back(wplist());
90  }
91  forget_weight(mine_[r]);
92  if (!is_next_bucket_full(r)) {
93  if (!reach_forgetting_threshold(r + 1) ||
94  mine_[r].size() == get_mine().size()) {
95  concat(mine_[r], mine_[r + 1]);
96  mine_[r].clear();
97  } else {
98  mine_[r + 1].swap(mine_[r]);
99  mine_[r].clear();
100  }
101  } else {
102  wplist cr = mine_[r];
103  wplist crr = mine_[r + 1];
104  mine_[r].clear();
105  mine_[r + 1].clear();
106  concat(cr, crr);
107  size_t dstsize = (r == 0) ? config_.compressed_bucket_size :
108  2 * r * r * config_.compressed_bucket_size;
109  compressor_->compress(crr, config_.bicriteria_base_size,
110  dstsize, mine_[r + 1]);
111  carry_up(r + 1);
112  }
113 }
