252  void get_row(const uint64_t id, framework::packer& pk) const {
253  jubatus::util::concurrent::scoped_rlock lk(table_lock_);
254  JUBATUS_ASSERT_GE(tuples_, id, "specified index is bigger than table size");
255  pk.pack_array(3); // [key, [owner, id], [data]]
256  pk.pack(keys_[id]); // key
257  pk.pack(versions_[id]); // [version]
258  pk.pack_array(columns_.size());
259  for (size_t i = 0; i < columns_.size(); ++i) {
260  columns_[i].pack_with_index(id, pk);
261  }
262  }

264  version_t set_row(const msgpack::object& o) {
265  if (o.type != msgpack::type::ARRAY || o.via.array.size != 3) {
266  throw msgpack::type_error();
267  }
268 
269  const std::string& key = o.via.array.ptr[0].as<std::string>();
270  version_t set_version = o.via.array.ptr[1].as<version_t>();
271 
272  jubatus::util::concurrent::scoped_wlock lk(table_lock_);
273  const msgpack::object& dat = o.via.array.ptr[2];
274  index_table::iterator it = index_.find(key);
275  if (it == index_.end()) { // did not exist, append
276  if (dat.via.array.size != columns_.size()) {
277  throw std::bad_cast();
278  }
279 
280  // add tuple
281  keys_.push_back(key);
282  versions_.push_back(set_version);
283  for (size_t i = 0; i < columns_.size(); ++i) {
284  columns_[i].push_back(dat.via.array.ptr[i]);
285  }
286  JUBATUS_ASSERT_EQ(keys_.size(), versions_.size(), "");
287 
288  // make index
289  index_.insert(std::make_pair(key, tuples_));
290  ++tuples_;
291  } else { // already exist, overwrite if needed
292  const uint64_t target = it->second;
293 
294  if (dat.via.array.size != columns_.size()) {
295  throw std::bad_cast();
296  }
297 
298  // overwrite tuple if needed
299  if (versions_[target].second <= set_version.second) {
300  // needed!!
301  versions_[target] = set_version;
302  for (size_t i = 0; i < columns_.size(); ++i) {
303  columns_[i].update(target, dat.via.array.ptr[i]);
304  }
305  // make index
306  index_.insert(std::make_pair(key, tuples_));
307  }
308  }
309  if (clock_ <= set_version.second) {
310  clock_ = set_version.second + 1;
311  }
312  return set_version;
313  }

391  void delete_row_(uint64_t index) {

416  ++clock_;
