36 void exp3::calc_weights_(const std::string& player_id,
37  std::vector<double>& weights) const {
38  const std::vector<std::string>& arms = s_.get_arm_ids();
39  if (arms.empty()) {
40  throw JUBATUS_EXCEPTION(
41  common::exception::runtime_error("arm is not registered"));
42  }
43 
44  const size_t n = arms.size();
45  weights.clear();
46  weights.reserve(n);
47  double total_weight = 0;
48  for (size_t i = 0; i < n; ++i) {
49  const double weight = std::exp(s_.get_arm_info(player_id, arms[i]).weight);
50  weights.push_back(weight);
51  total_weight += weight;
52  }
53  for (size_t i = 0; i < n; ++i) {
54  weights[i] = (1.0 - gamma_) * weights[i] / total_weight + gamma_ * n;
55  }
56 }

77 bool exp3::register_reward(const std::string& player_id,
78  const std::string& arm_id,
79  double reward) {
80  const std::vector<std::string>& arms = s_.get_arm_ids();
81  size_t i = std::find(arms.begin(), arms.end(), arm_id) - arms.begin();
82  if (i >= arms.size()) {
83  return false;
84  }
85  std::vector<double> weights;
86  calc_weights_(player_id, weights);
87  return s_.register_reward(player_id, arm_id,
88  reward * weights[i] * gamma_ / arms.size());
