62 splitter_weight_type make_weight_type(
63  const std::string& sample, const std::string& global) {
64  frequency_weight_type sample_type;
65  if (sample == "bin") {
66  sample_type = FREQ_BINARY;
67  } else if (sample == "tf") {
68  sample_type = TERM_FREQUENCY;
69  } else if (sample == "log_tf") {
70  sample_type = LOG_TERM_FREQUENCY;
71  } else {
72  throw JUBATUS_EXCEPTION(
73  converter_exception(std::string("unknown sample weight: ") + sample));
74  }
75 
76  term_weight_type global_type;
77  if (global == "bin") {
78  global_type = TERM_BINARY;
79  } else if (global == "idf") {
80  global_type = IDF;
81  } else if (global == "weight") {
82  global_type = WITH_WEIGHT_FILE;
83  } else {
84  throw JUBATUS_EXCEPTION(
85  converter_exception(std::string("unknown global weight: ") + global));
86  }
87  return splitter_weight_type(sample_type, global_type);

349 void initialize_converter(
350  const converter_config& config,
351  datum_to_fv_converter& conv,
352  const factory_extender* ext) {
353  using jubatus::util::lang::bind;
354  using jubatus::util::lang::_1;
355  using jubatus::util::lang::_2;
356 
357  if (config.hash_max_size.bool_test() && *config.hash_max_size.get() <= 0) {
358  std::stringstream msg;
359  msg << "hash_max_size must be positive, but is "
360  << *config.hash_max_size.get();
361  throw JUBATUS_EXCEPTION(converter_exception(msg.str()));
362  }
363 
364  std::map<std::string, string_filter_ptr> string_filters;
365  if (config.string_filter_types) {
366  string_filter_factory::create_function f;
367  if (ext) {
368  f = bind(&factory_extender::create_string_filter, ext, _1, _2);
369  }
370  init_string_filter_types(*config.string_filter_types, string_filters, f);
371  }
372 
373  std::map<std::string, num_filter_ptr> num_filters;
374  if (config.num_filter_types) {
375  num_filter_factory::create_function f;
376  if (ext) {
377  f = bind(&factory_extender::create_num_filter, ext, _1, _2);
378  }
379  init_num_filter_types(*config.num_filter_types, num_filters, f);
380  }
381 
382  std::map<std::string, string_feature_ptr> splitters;
383  register_default_string_types(splitters);
384  if (config.string_types) {
385  string_feature_factory::create_function f;
386  if (ext) {
387  f = bind(&factory_extender::create_string_feature, ext, _1, _2);
388  }
389  init_string_types(*config.string_types, splitters, f);
390  }
391 
392  std::map<std::string, num_feature_ptr> num_features;
393  register_default_num_types(num_features);
394  if (config.num_types) {
395  num_feature_factory::create_function f;
396  if (ext) {
397  f = bind(&factory_extender::create_num_feature, ext, _1, _2);
398  }
399  init_num_types(*config.num_types, num_features, f);
400  }
401 
402  std::map<std::string, binary_feature_ptr> binary_features;
403  if (config.binary_types) {
404  binary_feature_factory::create_function f;
405  if (ext) {
406  f = bind(&factory_extender::create_binary_feature, ext, _1, _2);
407  }
408  init_binary_types(*config.binary_types, binary_features, f);
409  }
410 
411  std::map<std::string, combination_feature_ptr> combination_features;
412  register_default_combination_types(combination_features);
413  if (config.combination_types) {
414  combination_feature_factory::create_function f;
415  if (ext) {
416  f = bind(&factory_extender::create_combination_feature, ext, _1, _2);
417  }
418  init_combination_types(*config.combination_types, combination_features, f);
419  }
420 
421  conv.clear_rules();
422  if (config.string_filter_rules) {
423  init_string_filter_rules(*config.string_filter_rules, string_filters, conv);
424  }
425  if (config.num_filter_rules) {
426  init_num_filter_rules(*config.num_filter_rules, num_filters, conv);
427  }
428  if (config.string_rules) {
429  init_string_rules(*config.string_rules, splitters, conv);
430  }
431  if (config.num_rules) {
432  init_num_rules(*config.num_rules, num_features, conv);
433  }
434  if (config.binary_rules) {
435  init_binary_rules(*config.binary_rules, binary_features, conv);
436  }
437  if (config.combination_rules) {
438  init_combination_rules(
439  *config.combination_rules,
440  combination_features, conv);
441  }
442 
443  if (config.hash_max_size.bool_test()) {
444  conv.set_hash_max_size(*config.hash_max_size.get());
445  }
446 }
