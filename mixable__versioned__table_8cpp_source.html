126 void mixable_versioned_table::mix(
127  const msgpack::object& obj,

129  internal_diff* diff_obj = dynamic_cast<internal_diff*>(ptr.get());
130  if (!diff_obj) {
131  throw JUBATUS_EXCEPTION(

135  if (obj.type != msgpack::type::RAW) {
136  throw JUBATUS_EXCEPTION(

138  }
139  msgpack::unpacked msg;
140  msgpack::unpack(&msg, obj.via.raw.ptr, obj.via.raw.size);
141  msgpack::object o = msg.get();
142  if (o.type != msgpack::type::ARRAY) {
143  throw JUBATUS_EXCEPTION(
144  core::common::exception::runtime_error("bad diff_object"));
145  }
146  for (size_t i = 0; i < o.via.array.size; i++) {
147  diff_obj->objs.push_back(o.via.array.ptr[i]);
148  }
149

151  shared_ptr<msgpack::zone> owner(msg.zone().release());
152  diff_obj->zones.push_back(owner);
153  }

172 void mixable_versioned_table::pull_impl(
173  const version_clock& vc, framework::packer& pk) const {
174 
175  model_ptr table = get_model();
176  const uint64_t table_size = table->size();
177 
178  size_t pack_size = 0;
179  for (uint64_t i = 0; i < table_size; ++i) {
180  const version_t version = table->get_version(i);
181  version_clock::const_iterator it = vc.find(version.first);
182  if (it == vc.end() || it->second < version.second) {
183  pack_size++;
184  }
185  }
186  pk.pack_array(pack_size);
187 
188  for (uint64_t i = 0; i < table_size; ++i) {
189  const version_t version = table->get_version(i);
190  version_clock::const_iterator it = vc.find(version.first);
191  if (it == vc.end() || it->second < version.second) {
192  table->get_row(i, pk);
193  }
194  }
195 }

210 void mixable_versioned_table::update_version(const version_t& version) {
211  version_clock::iterator it = vc_.find(version.first);
212  if (it == vc_.end()) {
213  vc_.insert(version);
