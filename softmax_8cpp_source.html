38 std::string softmax::select_arm(const std::string& player_id) {
39  const std::vector<std::string>& arms = s_.get_arm_ids();
40  if (arms.empty()) {
41  throw JUBATUS_EXCEPTION(
42  common::exception::runtime_error("arm is not registered"));
43  }
44 
45  std::vector<double> weights;
46  weights.reserve(arms.size());
47 
48  for (size_t i = 0; i < arms.size(); ++i) {
49  double expectation = s_.get_expectation(player_id, arms[i]);
50  weights.push_back(std::exp(expectation / tau_));
51  }
52  return arms[select_by_weights(weights, rand_)];
53 }
