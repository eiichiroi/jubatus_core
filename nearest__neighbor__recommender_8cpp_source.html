99 void nearest_neighbor_recommender::update_row(
100  const std::string& id,
101  const common::sfv_t& diff) {
102  if (unlearner_) {
103  if (!unlearner_->touch(id)) {
104  throw JUBATUS_EXCEPTION(common::exception::runtime_error(
105  "no more space available to add new ID: " + id));
106  }
107  }
108  orig_.set_row(id, diff);
109  common::sfv_t row;
110  orig_.get_row(id, row);
