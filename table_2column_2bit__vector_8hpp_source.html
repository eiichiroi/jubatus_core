242  if (bit_num() != bv.bit_num()) {

244  "calc_hamming_similarity(): bit_vector length unmatch! " +

246  jubatus::util::lang::lexical_cast<std::string>(bv.bit_num()));
247  }
248  if (is_empty() && bv.is_empty()) {
249  return 0;
250  } else if (is_empty()) {

252  } else if (bv.is_empty()) {

271  static uint64_t pop_count(uint64_t r) {
272  r = (r & 0x5555555555555555ULL) +
273  ((r >> 1) & 0x5555555555555555ULL);
274  r = (r & 0x3333333333333333ULL) +
275  ((r >> 2) & 0x3333333333333333ULL);
276  r = (r + (r >> 4)) & 0x0f0f0f0f0f0f0f0fULL;
277  r = r + (r >> 8);

339  void msgpack_pack(msgpack::packer<Buffer>& packer) const {
340  packer.pack_array(2);
341  packer.pack(static_cast<uint64_t>(bit_num_));
342  packer.pack_raw(used_bytes());
343  if (bits_) {

347  common::write_big_endian(bits_[i], buf);
348  packer.pack_raw_body(buf, BLOCKSIZE);
349  }
350  } else {

353  packer.pack_raw_body(&c, 1);
354  }
355  }

357  void msgpack_unpack(msgpack::object o) {
358  if (o.type != msgpack::type::ARRAY || o.via.array.size != 2) {
359  throw msgpack::type_error(); // like MSGPACK_DEFINE
360  }
361  const msgpack::object* objs = o.via.array.ptr;
362  const uint64_t bit_num = objs[0].as<uint64_t>();
363  const msgpack::type::raw_ref data = objs[1].as<msgpack::type::raw_ref>();
364  if (data.size != memory_size(bit_num)) {
365  throw bit_vector_unmatch_exception(
366  "msgpack_unpack(): invalid length of packed data: "
367  "expected: " +
368  jubatus::util::lang::lexical_cast<std::string>(bit_num_) +
369  ", got: " +
370  jubatus::util::lang::lexical_cast<std::string>(data.size));
371  }
372  std::vector<bit_base> buf;

374  buf.push_back(common::read_big_endian<bit_base>(&data.ptr[i]));

377  JUBATUS_ASSERT(own_ || bits_ == NULL);
378  }
