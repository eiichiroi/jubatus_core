61 shared_ptr<bandit_base> bandit_factory::create(
62  const std::string& name,
63  const common::jsonconfig::config& param) {
64  if (name == "epsilon_greedy") {
65  if (param.type() == json::json::Null) {
66  throw JUBATUS_EXCEPTION(
67  common::config_exception() << common::exception::error_message(
68  "parameter block is not specified in config"));
69  }
70  epsilon_greedy_config conf =
71  config_cast_check<epsilon_greedy_config>(param);
72  return shared_ptr<bandit_base>(new epsilon_greedy(conf.epsilon));
73  } else if (name == "ucb1") {
74  return shared_ptr<bandit_base>(new ucb1());
75  } else if (name == "softmax") {
76  if (param.type() == json::json::Null) {
77  throw JUBATUS_EXCEPTION(

79  "parameter block is not specified in config"));
80  }
81  softmax_config conf = config_cast_check<softmax_config>(param);
82  return shared_ptr<bandit_base>(new softmax(conf.tau));
83  } else if (name == "exp3") {
84  if (param.type() == json::json::Null) {
85  throw JUBATUS_EXCEPTION(
86  common::config_exception() << common::exception::error_message(
87  "parameter block is not specified in config"));
88  }
89  exp3_config conf = config_cast_check<exp3_config>(param);
90  return shared_ptr<bandit_base>(new exp3(conf.gamma));
91  } else {
92  throw JUBATUS_EXCEPTION(
93  common::unsupported_method("bandit(" + name + ")"));
94  }
